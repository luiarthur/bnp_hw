\frame{ %2
  \frametitle{Model}
  \vspace{0mm}
  \[
    \begin{array}{rclcl}
      \m y &|& \m\theta,\beta,\tau^2 &\overset{ind.}{\sim}&\text{Normal}_n(\m\theta+\m x\beta,~\tau^2\m I_n) \\
      \m \theta &|& G^{(n)} &\sim& G^{(n)} \\
      G^{(n)} &|& \alpha, \sigma^2, \phi &\sim& \text{DP}(~\alpha,~\text{Normal}_n(\m 0_n,\sigma^2H_n(\phi)) ~) \\
              && \beta, \tau^2 &\sim& \text{Normal}(0,10^2) \times \text{Inverse-Gamma}(1,1) \\
              && \alpha &\sim& \text{Gamma}(1,1) \\
              && \sigma^2 &\sim& \text{Inverse-Gamma}(1,1) \\
              && \phi &\sim& \text{Gamma}(1,1) \\
    \end{array}
  \]
  where 
  \begin{itemize}
    \item $H_n(\phi)$ is the exponential covariance function with decay parameter $\phi$. (i.e. 
      $(H_n(\phi))_{ij} = \exp\left\{-\phi~\norm{s_i-s_j}\right\}$.) \\
    \item $\m y$ are the station-measured Ozone levels, and 
    \item $\m x$ are a weighted sum of the CMAQ values of the 10 locations nearest to the prediction location.
  \end{itemize}
}
